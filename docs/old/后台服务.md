```dart
  }

  }
```

```dart
@pragma('vm:entry-point')
void _onStart(ServiceInstance service) async {
  // 确保 Flutter 绑定初始化
  WidgetsFlutterBinding.ensureInitialized();
  DartPluginRegistrant.ensureInitialized();

  // 初始化依赖
  await _initializeDependencies();

  // 设置定时器
  Timer.periodic(const Duration(minutes: 1), (timer) async {
    await _checkAndTriggerActiveReply();
  });

  // 监听停止命令
  service.on('stop').listen((event) {
    service.stopSelf();
  });

  // 更新通知
  if (service is AndroidServiceInstance) {
    service.setForegroundNotificationInfo(
      title: 'LNPhone',
      content: '后台服务运行中',
    );
  }
}

Future<void> _initializeDependencies() async {
  // 初始化数据库（独立连接）
  final db = AppDatabase();
  
  // 初始化通知服务
  await NotificationService.initialize();
  
  // 加载提示词配置
  await PromptConfigLoader.initialize();
}
```

### 3.3 主动回复检查

```dart
Future<void> _checkAndTriggerActiveReply() async {
  try {
    final db = AppDatabase();

    // 1. 检查全局开关
    final activeReplyEnabled = await db.getSetting('active_reply_enabled');
    if (activeReplyEnabled != 'true') {
      return;
    }

    // 2. 检查用户不活跃时间
    final inactivityMinutesStr = await db.getSetting('active_reply_inactivity_minutes');
    final inactivityMinutes = int.tryParse(inactivityMinutesStr ?? '30') ?? 30;

    final lastActiveStr = await db.getSetting('last_user_active_time');
    final lastActive = int.tryParse(lastActiveStr ?? '0') ?? 0;
    final now = DateTime.now().millisecondsSinceEpoch;

    if (now - lastActive < inactivityMinutes * 60 * 1000) {
      return; // 用户仍在活跃
    }

    // 3. 获取所有会话
    final sessions = await db.getAllChatSessions();

    for (final session in sessions) {
      await _checkAndTriggerForSession(db, session, now);
    }
  } catch (e) {
    AppLogService.e('Background check failed: $e');
  }
}

Future<void> _checkAndTriggerForSession(
  AppDatabase db,
  ChatSession session,
  int now,
) async {
  // 检查会话级别开关
  // 注意：会话可能没有单独的开关，使用全局开关

  // 获取最后一条消息
  final lastMessage = await db.getLastMessage(session.id);
  if (lastMessage == null) return;

  // 检查最后消息时间
  final replyIntervalStr = await db.getSetting('active_reply_interval_minutes');
  final replyInterval = int.tryParse(replyIntervalStr ?? '60') ?? 60;

  if (now - lastMessage.timestamp < replyInterval * 60 * 1000) {
    return; // 距离上次消息时间不够
  }

  // 检查是否是 AI 消息（避免重复主动回复）
  if (!lastMessage.isMe) {
    // 最后一条是 AI 消息，检查是否需要再次主动回复
    final lastTriggerStr = await db.getSetting('last_active_reply_${session.id}');
    final lastTrigger = int.tryParse(lastTriggerStr ?? '0') ?? 0;

    if (now - lastTrigger < replyInterval * 60 * 1000) {
      return; // 距离上次主动回复时间不够
    }
  }

  // 触发 AI 回复
  await _triggerAiReply(db, session);
}
```

### 3.4 AI 回复触发

```dart
Future<void> _triggerAiReply(AppDatabase db, ChatSession session) async {
  try {
    // 1. 获取角色和用户信息
    final role = await db.getContactRole(session.roleId);
    final me = await db.getContactMe(session.meId);
    if (role == null || me == null) return;

    // 2. 获取 API 预设
    final apiPresetId = session.apiPresetId;
    if (apiPresetId == null) return;
    final apiPreset = await db.getApiPreset(apiPresetId);
    if (apiPreset == null) return;

    // 3. 获取历史消息（最近 N 条）
    final history = await db.getRecentMessages(session.id, limit: 50);

    // 4. 获取记忆和世界书
    final memories = await db.getMemoriesForRole(session.roleId);
    final worldInfos = await db.getWorldInfosByIds(session.worldInfoIds);
    final textPresets = await db.getTextPresetsByIds(session.textPresetIds);

    // 5. 获取可用表情（如果启用）
    List<EmojiModel> emojis = [];
    if (session.enableEmoji) {
      emojis = await db.getAvailableEmojisForRole(session.roleId);
    }

    // 6. 构建提示词配置
    final promptConfig = await PromptConfigLoader.load();

    // 7. 添加主动回复指令
    final activeReplyPrompt = '''
当前情况：用户已经有一段时间没有回复了。
作为 ${role.name}，你可以主动发起话题，比如：
- 关心用户在做什么
- 分享一些有趣的事情
- 询问之前话题的后续
- 发送一个表情表达想念

请自然地发起对话，不要显得太刻意。
''';

    // 8. 调用 LLM
    final messages = await LlmService.generateResponse(
      apiPreset: apiPreset,
      promptConfig: promptConfig,
      history: history,
      role: role,
      me: me,
      memories: memories,
      worldInfos: worldInfos,
      textPresets: textPresets,
      enableExtendedChat: session.enableExtendedChat,
      enableEmoji: session.enableEmoji,
      availableEmojis: emojis,
      additionalSystemPrompt: activeReplyPrompt,
    );

    if (messages.isEmpty) return;

    // 9. 保存消息
    for (final message in messages) {
      final fullMessage = message.copyWith(
        id: '${DateTime.now().millisecondsSinceEpoch}-${Random().nextInt(10000)}',
        sessionId: session.id,
        isMe: false,
        sender: role.name,
        timestamp: DateTime.now().millisecondsSinceEpoch,
        isRead: false,
      );
      await db.insertMessage(fullMessage);
    }

    // 10. 更新会话时间
    await db.updateSessionLastUpdated(session.id);

    // 11. 记录触发时间
    await db.setSetting(
      'last_active_reply_${session.id}',
      DateTime.now().millisecondsSinceEpoch.toString(),
    );

    // 12. 发送通知
    await NotificationService.showMessageNotification(
      title: role.name,
      body: _getNotificationBody(messages.first),
      sessionId: session.id,
    );

  } catch (e) {
    AppLogService.e('Trigger AI reply failed: $e');
  }
}

String _getNotificationBody(ChatMessage message) {
  switch (message.type) {
    case MessageType.words:
      return message.content.length > 50 
          ? '${message.content.substring(0, 50)}...' 
          : message.content;
    case MessageType.emoji:
      return '[表情]';
    case MessageType.image:
      return '[图片]';
    case MessageType.action:
      return message.content;
    default:
      return '发来一条消息';
  }
}
```

## 4. 通知服务

### 4.1 初始化

```dart
// lib/core/services/notification_service.dart

class NotificationService {
  static final FlutterLocalNotificationsPlugin _plugin = 
      FlutterLocalNotificationsPlugin();

  static Future<void> initialize() async {
    // Android 设置
    const androidSettings = AndroidInitializationSettings('@mipmap/ic_launcher');
    
    // iOS 设置
    const iosSettings = DarwinInitializationSettings(
      requestAlertPermission: true,
      requestBadgePermission: true,
      requestSoundPermission: true,
    );

    const settings = InitializationSettings(
      android: androidSettings,
      iOS: iosSettings,
    );

    await _plugin.initialize(
      settings,
      onDidReceiveNotificationResponse: _onNotificationTapped,
    );

    // 创建通知渠道
    await _createNotificationChannels();
  }

  static Future<void> _createNotificationChannels() async {
    final androidPlugin = _plugin.resolvePlatformSpecificImplementation<
        AndroidFlutterLocalNotificationsPlugin>();

    if (androidPlugin != null) {
      // 消息通知渠道
      await androidPlugin.createNotificationChannel(
        const AndroidNotificationChannel(
          'messages',
          '消息通知',
          description: '聊天消息通知',
          importance: Importance.high,
          sound: RawResourceAndroidNotificationSound('notification_tone'),
        ),
      );

      // 后台服务渠道
      await androidPlugin.createNotificationChannel(
        const AndroidNotificationChannel(
          'lnphone_background',
          '后台服务',
          description: '后台服务运行状态',
          importance: Importance.low,
        ),
      );
    }
  }
}
```

### 4.2 显示通知

```dart
static Future<void> showMessageNotification({
  required String title,
  required String body,
  required String sessionId,
}) async {
  // 获取自定义提示音设置
  final db = AppDatabase();
  final customSound = await db.getSetting('notification_sound');

  final androidDetails = AndroidNotificationDetails(
    'messages',
    '消息通知',
    channelDescription: '聊天消息通知',
    importance: Importance.high,
    priority: Priority.high,
    sound: customSound != null 
        ? RawResourceAndroidNotificationSound(customSound)
        : null,
    styleInformation: BigTextStyleInformation(body),
  );

  const iosDetails = DarwinNotificationDetails(
    presentAlert: true,
    presentBadge: true,
    presentSound: true,
  );

  final details = NotificationDetails(
    android: androidDetails,
    iOS: iosDetails,
  );

  // 使用 sessionId 的 hashCode 作为通知 ID
  // 这样同一会话的通知会更新而非堆叠
  await _plugin.show(
    sessionId.hashCode,
    title,
    body,
    details,
    payload: jsonEncode({'sessionId': sessionId}),
  );
}
```

### 4.3 通知点击处理

```dart
static void _onNotificationTapped(NotificationResponse response) {
  if (response.payload == null) return;

  try {
    final data = jsonDecode(response.payload!);
    final sessionId = data['sessionId'] as String?;

    if (sessionId != null) {
      // 通过全局导航跳转到对应会话
      NavigationService.navigateToChatScreen(sessionId);
    }
  } catch (e) {
    AppLogService.e('Parse notification payload failed: $e');
  }
}
```

## 5. 主动回复机制

### 5.1 配置项

| 设置项 | 类型 | 默认值 | 说明 |
|--------|------|--------|------|
| `active_reply_enabled` | bool | false | 全局开关 |
| `active_reply_inactivity_minutes` | int | 30 | 用户不活跃多久后触发 |
| `active_reply_interval_minutes` | int | 60 | 主动回复最小间隔 |
| `last_user_active_time` | int | - | 用户最后活跃时间戳 |

### 5.2 触发条件

```
┌─────────────────────────────────────────────────────────────┐
│                     主动回复触发条件                          │
│                                                              │
│  全局条件:                                                    │
│  ├── active_reply_enabled == true                           │
│  └── (当前时间 - last_user_active_time) > inactivity_minutes │
│                                                              │
│  会话条件:                                                    │
│  ├── 会话存在 API 预设                                        │
│  ├── (当前时间 - 最后消息时间) > interval_minutes             │
│  └── 如果最后是 AI 消息:                                      │
│      └── (当前时间 - 上次触发时间) > interval_minutes         │
│                                                              │
└─────────────────────────────────────────────────────────────┘
```

## 7. 设计决策

### 7.1 为什么使用 Foreground Service？

| 方案 | 优点 | 缺点 | 选择 |
|------|------|------|------|
| 普通后台服务 | 简单 | 容易被系统杀死 | ❌ |
| Foreground Service | 稳定、持久 | 需要显示通知 | ✅ |
| WorkManager | 系统调度、省电 | 无法精确控制时间 | ❌ |
| AlarmManager | 精确定时 | 需要额外处理 | ❌ |

**选择理由**: Foreground Service 提供最稳定的后台运行能力，虽然需要显示持久通知，但对于需要主动回复的应用来说是可接受的。

### 7.2 为什么使用独立数据库连接？

**问题**: 后台服务运行在独立 Isolate，无法共享主进程的数据库连接。

**解决方案**: 在后台 Isolate 中创建独立的数据库连接。

```dart
// 后台 Isolate
void _onStart(ServiceInstance service) async {
  final db = AppDatabase(); // 新的连接
  // ...
}
```

**注意事项**:
- SQLite 支持多连接并发读取
- 写入时需要处理锁冲突
- 使用 WAL 模式提高并发性能

### 7.3 为什么直接发送通知？

**问题**: 后台 Isolate 与主进程通信复杂

**方案对比**:

| 方案 | 复杂度 | 可靠性 |
|------|--------|--------|
| IPC 通知主进程 | 高 | 低（主进程可能被杀） |
| 直接发送通知 | 低 | 高 |

**选择**: 后台服务直接调用通知 API，无需通过主进程。

### 7.4 主动回复的限制

为避免打扰用户，设置了多重限制：

1. **全局开关**: 用户可完全关闭
2. **时间阈值**: 只在用户长时间不活跃时触发
3. **间隔限制**: 同一会话不会频繁触发
4. **智能判断**: 如果最后是 AI 消息，增加额外延迟
